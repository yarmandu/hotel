-- Proc para Insertar, Actualizar CLIENTE
use ProyectoHoteleria;
DELIMITER //
CREATE PROCEDURE uspClienteInsertar	(	IN v_IdCliente INT,
								IN v_IdTipoDocumento INT,
								IN v_descTipoDoc VARCHAR(15), 
								IN v_Nombres VARCHAR(100), 
                                IN v_ApePaterno VARCHAR(50), 
                                IN v_ApeMaterno VARCHAR(50), 
								IN v_Direccion VARCHAR(200), 
                                IN v_Telefono VARCHAR(20), 
                                IN v_IdTipoSexo INT, 
								IN v_IdTipoCliente INT,
                                IN v_IdTipoResidencia INT)
BEGIN
	DECLARE v_salida int;
	DECLARE v_IdClienteExist INT;
	SET v_salida = 0;
	IF v_IdCliente = 0 then
			SET v_IdClienteExist = (SELECT C.IdCliente
									FROM Cliente AS C 
									WHERE C.IdTipoDocumento = v_IdTipoDocumento and C.descTipoDoc = v_descTipoDoc and C.estado = 0);
			IF (v_IdClienteExist <> 0 or v_IdClienteExist <> null) THEN
					UPDATE Cliente 
					SET Nombres				= 	v_Nombres
						,ApePaterno			=	v_ApePaterno
						,ApeMaterno			=	v_ApeMaterno
						,Direccion			=	v_Direccion
						,Telefono			=	v_Telefono
						,IdTipoSexo			=	v_IdTipoSexo
						,IdTipoCliente		=	v_IdTipoCliente
						,IdTipoResidencia	=	v_IdTipoResidencia
                        ,estado				=	1
					WHERE IdCliente 		=	v_IdClienteExist;
					SET v_salida = 1;
					SELECT v_salida;
			ELSE
					INSERT INTO Cliente(IdTipoDocumento, 
										descTipoDoc,
										Nombres,
										ApePaterno,
                                        ApeMaterno,
                                        Direccion,
                                        Telefono,
                                        IdTipoSexo,
                                        IdTipoCliente,
                                        IdTipoResidencia,
                                        estado)
					VALUES (v_IdTipoDocumento,
							v_descTipoDoc,
							v_Nombres,
                            v_ApePaterno,
							v_ApeMaterno,
                            v_Direccion,
                            v_Telefono,
                            v_IdTipoSexo,
                            v_IdTipoCliente,
                            v_IdTipoResidencia,
                            1);
					set v_salida = 1;
					SELECT v_salida;
			END IF;	
	ELSE
		UPDATE Cliente 
		SET IdTipoDocumento		= 	v_IdTipoDocumento
			,descTipoDoc		=	v_descTipoDoc
			,Nombres			= 	v_Nombres
			,ApePaterno			=	v_ApePaterno
			,ApeMaterno			=	v_ApeMaterno
			,Direccion			=	v_Direccion
			,Telefono			=	v_Telefono
			,IdTipoSexo			=	v_IdTipoSexo
			,IdTipoCliente		=	v_IdTipoCliente
			,IdTipoResidencia	=	v_IdTipoResidencia
			,estado				=	1
		WHERE IdCliente 		=	v_IdCliente;
		SET v_salida = 2;
		SELECT v_salida;
	END IF;
END//

DELIMITER ;

CREATE PROCEDURE uspinsertarHabitacion	(	IN v_NroHabitacion INT,
										IN v_idTipoHabitacion INT,
										IN v_idtipoServicio INT, 
										IN v_precio decimal, 
										IN v_cantCamas INT)
BEGIN
DECLARE v_salida int;
	SET v_salida = 0;
	IF exists (SELECT C.nroHabitacion 
						FROM Habitacion AS C 
                        WHERE C.nroHabitacion = v_NroHabitacion) then
		BEGIN										
						UPDATE Habitacion 
							SET idTipoHabitacion		= v_idTipoHabitacion
								,idTipoServicio		=	v_idtipoServicio
								,Precio				= 	v_precio
								,cantCama			=	v_cantCamas
								,estado				=	1
							WHERE nroHabitacion 	=	v_NroHabitacion;
							SET v_salida = 2;
						SELECT v_salida;
				
		END;
	ELSE
		BEGIN
				INSERT INTO Habitacion(nroHabitacion,
										idTipoHabitacion, 
										idTipoServicio,
										Precio,
										cantCama,
										estado)
								VALUES (v_NroHabitacion,
										v_idTipoHabitacion,
										v_idtipoServicio,
										v_precio,
										v_cantCamas,
										1);
							set v_salida = 1;
							SELECT v_salida;
		END;
	END IF;
END//

DELIMITER ;

	select r.idReserva,r.IdHotel, r.descTipoDoc, r.cantHabitacion, 
    r.idTipoHabitacion, th.tipoHabitacion, 
    r.fechaInicio, r.fechaFin,r.IdTipoDocumento, td.tipoDocumento,
    r.idtipoPago, tp.tipoPago, r.Precio,r.precioTotal,r.total,r.estado
    from ReservaHabitacion r
    inner join TipoPago tp
    on r.idtipoPago = tp.idtipoPago
    inner join TipoHabitacion th
    on r.idTipoHabitacion = th.idTipoHabitacion
    inner join TipoDocumento td
    on r.IdTipoDocumento = td.IdTipoDocumento
    where (r.idReserva = v_idReserva) and 
    (r.estado = 1);
END//

CREATE PROCEDURE `uspAllHotel`(IN v_IdDistrito int, IN v_Inicio int, IN v_Tamano int)
BEGIN
	SELECT h.intIdHotel IdHotel, 
	IFNULL(h.varNombre,'') Nombre, 
	IFNULL(h.varDescripcion, '') Descripcion, 
	IFNULL(h.varTelefono, '') Telefono, 
	CONCAT(IFNULL(h.varDireccion, ''), ' - ', d.varDescripcion) Direccion, 
	IFNULL(h.varHorario, '') Horario, 
	IFNULL(h.varImagen, '') Imagen, 
	IFNULL(h.varLongitud, 0) Longitud, 
	IFNULL(h.varLatitud, 0) Latitud,
	'Si' Reserva
	FROM  	tblHotel h
	INNER 	JOIN  tblDistrito d ON h.intIdDistrito = d.intIdDistrito 
	WHERE  	h.bitEstado = 1 and h.intIdDistrito = v_IdDistrito
	ORDER BY Nombre, IdHotel
	Limit v_Inicio, v_Tamano;
END//

DELIMITER ;

CREATE PROCEDURE `uspAllGpsHotel`(IN v_Longitud decimal(20,6), IN v_Latitud decimal(20,6), IN v_Distancia int, IN v_Tamano int, IN v_Pagina int)
BEGIN
DECLARE v_Inicio INT;
	set v_Inicio = (v_Pagina - 1) * v_Tamano;
	SELECT * FROM (
	SELECT h.intIdHotel IdHotel, 
	IFNULL(h.varNombre,'') Nombre,
	IFNULL(h.varDescripcion, '') Descripcion,
	IFNULL(h.varTelefono, '') Telefono, 
	CONCAT(IFNULL( h.varDireccion,''), ' - ', d.varDescripcion) Direccion,
	IFNULL(h.varHorario, '') Horario,
	h.varImagen Imagen,	
	( 6371 * ACOS( SIN( RADIANS( h.varLatitud ) ) 
	* SIN( RADIANS( v_Latitud ) ) + COS( RADIANS( h.varLongitud - v_Longitud ) ) * COS( RADIANS( h.varLatitud ) ) 
	* COS( RADIANS( v_Latitud ) ) ) ) AS DistanciaHotel,
	h.varLongitud Longitud,
	h.varLatitud Latitud,
	'Si' Reserva
	FROM tblHotel h
	INNER JOIN  tblDistrito d ON h.intIdDistrito = d.intIdDistrito
	WHERE  h.bitEstado = 1) HOTEL
	HAVING DistanciaHotel < v_Distancia
	ORDER BY Nombre, IdHotel								
	Limit v_Inicio, v_Tamano;
END//

DELIMITER ;

CREATE PROCEDURE `uspAllUbigeo`()

	SELECT 1 flag,intIdPais Id,varDescripcion Descripcion,intCodPais Codigo,varZonaGeografica ZonaGeografica 
	FROM proyectohoteleria.tblPais where bitEstado=1
	UNION
	SELECT 2,a.intIdDepartamento IdDepartamento, a.varDescripcion Descripcion, a.intIdPais IdPais,' - '
	FROM proyectohoteleria.tblDepartamento a
	WHERE a.bitEstado=1
	UNION
	SELECT 3, a.intIdProvincia IdProvincia, a.varDescripcion Descripcion, a.intIdDepartamento IdDepartamento,' - '
	FROM proyectohoteleria.tblProvincia a 
	WHERE a.bitEstado=1
	UNION
	SELECT 4, a.intIdDistrito IdDistrito, a.varDescripcion Descripcion, a.intIdProvincia IdProvincia,' - '
	FROM proyectohoteleria.tblDistrito a
	WHERE a.bitEstado=1;
